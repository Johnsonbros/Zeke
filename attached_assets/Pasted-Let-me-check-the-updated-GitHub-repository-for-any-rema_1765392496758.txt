import os

SEARCH_PATTERNS = [

    (r'api\.limitless\.ai', 'Limitless API URL'),
    (r'/v1/lifelogs', 'Limitless lifelog endpoint'),
    (r'/v1/chats', 'Limitless chat endpoint'),
    (r'ContentNode', 'Limitless ContentNode class'),
    (r'LIMITLESS_API_KEY', 'Limitless API key'),
    (r'startOffsetMs|endOffsetMs', 'Limitless timestamp fields'),
]

def should_skip(path):
    parts = Path(path).parts
    return any(skip in parts for skip in SKIP_DIRS) or \
           Path(path).suffix in SKIP_EXTENSIONS

def scan_file(filepath):
    findings = []
    try:
        with open(filepath, 'r', encoding='utf-8', errors='ignore') as f:
            content = f.read()
            for pattern, description in SEARCH_PATTERNS:
                matches = re.finditer(pattern, content, re.IGNORECASE)
                for match in matches:
                    line_num = content[:match.start()].count('\n') + 1
                    line_content = content.split('\n')[line_num - 1].strip()
                    findings.append({
                        'pattern': description,
                        'line': line_num,
                        'content': line_content
                    })
    except Exception as e:
        pass
    return findings

def main():
    print("üîç SCANNING FOR LIMITLESS REFERENCES...")
    print("=" * 60)
    
    total_findings = 0
    files_with_issues = {}
    
    for root, dirs, files in os.walk('.'):
        # Remove skip directories from walk
        dirs[:] = [d for d in dirs if d not in SKIP_DIRS]
        
        for file in files:

            if should_skip(filepath):
                continue
                
            findings = scan_file(filepath)
            if findings:
                files_with_issues[filepath] = findings
                total_findings += len(findings)
    
    # Print results
    if files_with_issues:
        print(f"\n‚ùå FOUND {total_findings} LIMITLESS REFERENCES IN {len(files_with_issues)} FILES:\n")

    else:

    # Check for specific file

if __name__ == "__main__":
    main()
